
#include <stdio.h>

int main()
{

    for(i=0;i<N;i++)

        scanf("%d",&a[i]);
    }
    printf("\nthe array entered:\t");
    for(i=0;i<N;i++)

        printf("%d\t",a[i]);
    }
    printf("\nEnter the index location of element to be deleted:");
    scanf("%d",&n);

    for(i=n-1;i<N-1;i++)
